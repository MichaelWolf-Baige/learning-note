## 第1题 互斥锁

```c
// ——— 全局 ———
static int g_ticket = 100;

// ——— 抢票线程 ———
void Worker(void *arg) {
    osMutexId_t *mid = (osMutexId_t *)arg;
    while (1) {
        osMutexAcquire(*mid, osWaitForever);   // 上锁
        if (g_ticket > 0) { g_ticket--; osMutexRelease(*mid); }
        else             { osMutexRelease(*mid); break; }
    }
}

// ——— 主函数 ———
static void MutexMain(void *arg) {
    (void)arg;
    osMutexId_t mid = osMutexNew(&(osMutexAttr_t){0});
    newThread("w1", Worker, &mid);
    newThread("w2", Worker, &mid);
    newThread("w3", Worker, &mid);
    osDelay(30000);
}
```

## 第2题 信号量

```c
// ——— 全局 ———
static osSemaphoreId_t g_sem;

// ——— 工人线程 ———
void Worker(void *arg) {
    int *t = (int *)arg;
    while (1) {
        osSemaphoreAcquire(g_sem, osWaitForever);  // P操作
        osDelay(*t);
        osSemaphoreRelease(g_sem);                  // V操作
        osDelay(100);
    }
}

// ——— 主函数 ———
static void SemMain(void *arg) {
    (void)arg;
    g_sem = osSemaphoreNew(3, 3, NULL);    // 最大3，初始3
    int t1=200, t2=300, t3=400, t4=500;
    newThread("w1", Worker, &t1);
    newThread("w2", Worker, &t2);
    newThread("w3", Worker, &t3);
    newThread("w4", Worker, &t4);
    osDelay(10000);
}
```

## 第3题 队列

```c
// ——— 全局 ———
typedef struct { char letter; int sender_id; } Msg_t;
static osMessageQueueId_t g_q;

// ——— 发送 ———
void Sender(void *arg) {
    int id = *(int*)arg; Msg_t m;
    for (char c='a'; c<='z'; c++) {
        m.letter=c; m.sender_id=id;
        osMessageQueuePut(g_q, &m, 0, osWaitForever);
        osDelay(100);
    }
}

// ——— 接收 ———
void Receiver(void *arg) {
    Msg_t m;
    while (1) {
        osMessageQueueGet(g_q, &m, NULL, osWaitForever);
        if (m.letter == 'z') break;
    }
}

// ——— 主函数 ———
static void QueueMain(void *arg) {
    (void)arg;
    g_q = osMessageQueueNew(10, sizeof(Msg_t), NULL);
    int r1=1,r2=2,r3=3, s1=1,s2=2;
    newThread("R1",Receiver,&r1); newThread("R2",Receiver,&r2); newThread("R3",Receiver,&r3);
    newThread("S1",Sender,&s1);   newThread("S2",Sender,&s2);
    osDelay(5000);
    osMessageQueueDelete(g_q);
}
```

## 第4题 按键蜂鸣器

> 加头文件：`iot_gpio.h` `iot_pwm.h` `hi_io.h` `hi_pwm.h`

```c
#define PIN_KEY  5        // 按键
#define PIN_BUZ  9        // 蜂鸣器

static void BuzzerMain(void *arg) {
    (void)arg;
    // 初始化
    IoTGpioInit(PIN_KEY);  hi_io_set_func(PIN_KEY, HI_IO_FUNC_GPIO_5_GPIO);      IoTGpioSetDir(PIN_KEY, IOT_GPIO_DIR_IN);
    IoTGpioInit(PIN_BUZ);  hi_io_set_func(PIN_BUZ, HI_IO_FUNC_GPIO_9_PWM0_OUT); IoTGpioSetDir(PIN_BUZ, IOT_GPIO_DIR_OUT);

    while (1) {
        int v; IoTGpioGetInputVal(PIN_KEY, &v);
        if (v==0) IoTPwmStart(0, 20, 4000);   // 按下→响
        else      IoTPwmStop(0);
        osDelay(20);
    }
}
```

## 第5题 交通灯

> 头文件同第4题
> **记 PWM 端口对应：** PWM0=蜂鸣器, PWM1=红(GPIO10), PWM2=绿(GPIO11), PWM3=黄(GPIO12)

```c
#define R  HI_IO_NAME_GPIO_10
#define G  HI_IO_NAME_GPIO_11
#define Y  HI_IO_NAME_GPIO_12
#define BZ HI_IO_NAME_GPIO_9

static void TrafficMain(void *arg) {
    (void)arg;
    // 初始化（4个GPIO三行一套：Init→set_func→SetDir）
    IoTGpioInit(R);  hi_io_set_func(R,  HI_IO_FUNC_GPIO_10_PWM1_OUT); IoTGpioSetDir(R,  IOT_GPIO_DIR_OUT);
    IoTGpioInit(G);  hi_io_set_func(G,  HI_IO_FUNC_GPIO_11_PWM2_OUT); IoTGpioSetDir(G,  IOT_GPIO_DIR_OUT);
    IoTGpioInit(Y);  hi_io_set_func(Y,  HI_IO_FUNC_GPIO_12_PWM3_OUT); IoTGpioSetDir(Y,  IOT_GPIO_DIR_OUT);
    IoTGpioInit(BZ); hi_io_set_func(BZ, HI_IO_FUNC_GPIO_9_PWM0_OUT);  IoTGpioSetDir(BZ, IOT_GPIO_DIR_OUT);
    IoTPwmInit(0); IoTPwmInit(1); IoTPwmInit(2); IoTPwmInit(3);

    while (1) {
        IoTPwmStart(1,90,4000); IoTPwmStart(0,20,4000); osDelay(500); IoTPwmStop(0); osDelay(500);  // 红5s
        IoTPwmStart(2,90,4000); IoTPwmStart(0,20,4000); osDelay(200); IoTPwmStop(0); osDelay(200);  // 绿5s
        IoTPwmStart(3,90,4000); IoTPwmStart(0,20,4000); osDelay(500); IoTPwmStop(0); osDelay(500);  // 黄2s
    }
}
```

## 第10题 OLED 滚动

> 加头文件：`ssd1306.h` `string.h`

```c
#define STU_ID   "[你的学号]"
#define NAME_PY  "[你的拼音]"
char g_buf[256]; int g_len;

static void OledMain(void *arg) {
    (void)arg;
    ssd1306_Init();
    sprintf(g_buf, "%s%sabcdefghijklmnopqrstuvwxyz", STU_ID, NAME_PY);
    g_len = strlen(g_buf);
    int off = 0;
    while (1) {
        char s[17]={0};
        for (int i=0;i<16;i++) s[i] = g_buf[(off+i) % g_len];
        ssd1306_Fill(Black); ssd1306_SetCursor(0,20);
        ssd1306_DrawString(s, Font_7x10, White); ssd1306_UpdateScreen();
        off++; osDelay(500);
    }
}
```

## 第12题 交通灯倒计时

> 头文件同第10题

```c
static void CountdownMain(void *arg) {
    (void)arg;
    ssd1306_Init();
    int state = 0;   // 0=绿 1=黄 2=红
    while (1) {
        int sec = (state==0)?20 : (state==1)?5 : 15;
        while (sec > 0) {
            char buf[32];
            ssd1306_Fill(Black); ssd1306_SetCursor(20,20);
            ssd1306_DrawString((state==0)?"Green":(state==1)?"Yellow":"Red", Font_7x10, White);
            snprintf(buf, sizeof(buf), "Time: %02d", sec);
            ssd1306_SetCursor(20,45); ssd1306_DrawString(buf, Font_7x10, White);
            ssd1306_UpdateScreen();
            osDelay(1000); sec--;
        }
        state = (state+1) % 3;
    }
}
```

# 记忆卡片（考前快速过一遍）

| 题 | 核心API/关键词 | 关键数字 |
|---|---|---|
| 1 互斥锁 | `osMutexNew` → `Acquire` → 改共享变量 → `Release` | 票数100, 线程×3 |
| 2 信号量 | `osSemaphoreNew(3,3)` → `Acquire` → `Delay(工时)` → `Release` | 3台机器, 4工人, 工时200/300/400/500 |
| 3 队列 | `osMessageQueueNew(10,sizeof)` → `Put`(a~z) → `Get`(z退出) | 队列长10, 3收2发 |
| 4 按键蜂鸣 | `IoTGpioInit(5)` → `GetInputVal` → `IoTPwmStart(0,20,4000)` | GPIO5按键, GPIO9蜂鸣, 20%占空比 |
| 5 交通灯 | GPIO10红/11绿/12黄/9蜂鸣 → `IoTPwmStart(n,90,4000)` | 红5s绿5s黄2s, 蜂鸣间隔不同 |
| 6 炫彩灯 | GPIO10/11/12 → `IoTPwmStart(1/2/3, i/j/k, 4000)` | 三层循环 1~99 |
| 7 小夜灯 | ADC3(人体)+ADC4(光线) → `>1200`→亮灯 | PWM65/39/8, 阈值1200 |
| 8 燃气报警 | ADC5 → `>600` → `IoTPwmStart(0,50,2700)` | 阈值600, 2700Hz |
| 9 温湿度 | I2C GPIO13(SDA)/14(SCL) → `AHT20_GetMeasureResult` → `humi>90` | I2C0, 400000波特率 |
| 10 OLED滚动 | `ssd1306_Init` → `sprintf`拼接 → 偏移取16字符→`DrawString` | 16字符窗口, 500ms间隔 |
| 11 按键移字符 | S1左移x-=7, S2右移x+=7 → `DrawChar` | 起始x=64, 边界0/121 |
| 12 交通灯倒计时 | OLED显示状态+倒计时 → `osDelay(1000)` → 状态轮转 | 绿20s→黄5s→红15s |
