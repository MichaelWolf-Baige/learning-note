## 1. 互斥锁 — 抢票（100张票，3线程）

```c
static int g_ticket = 100;

void number_thread(void *arg) {
    osMutexId_t *mid = (osMutexId_t *)arg;
    while (1) {
        osMutexAcquire(*mid, osWaitForever);
        if (g_ticket > 0) { g_ticket--; osMutexRelease(*mid); }
        else { osMutexRelease(*mid); break; }
    }
}

// 主函数: mid = osMutexNew(NULL); newThread×3
```

## 2. 信号量 — 工人用机器（3台机器4工人，工时200/300/400/500）

```c
static osSemaphoreId_t g_sem;

void worker_thread(void *arg) {
    int *t = (int *)arg;
    while (1) {
        osSemaphoreAcquire(g_sem, osWaitForever);
        osDelay(*t);
        osSemaphoreRelease(g_sem);
        osDelay(100);
    }
}

// 主函数: g_sem = osSemaphoreNew(3,3,NULL); newThread×4 传各工时
```

## 3. 消息队列 — a~z（队列10，2发3收，收到z退出）

```c
typedef struct { char letter; int sender_id; } Msg_t;
static osMessageQueueId_t g_q;

void sender_thread(void *arg) {
    int id = *(int*)arg; Msg_t m;
    for (char c='a'; c<='z'; c++) {
        m.letter=c; m.sender_id=id;
        osMessageQueuePut(g_q, &m, 0, osWaitForever);
        osDelay(100);
    }
}

void receiver_thread(void *arg) {
    Msg_t m;
    while (1) {
        osMessageQueueGet(g_q, &m, NULL, osWaitForever);
        if (m.letter == 'z') break;
    }
}

// 主函数: g_q = osMessageQueueNew(10,sizeof(Msg_t),NULL); newThread收×3发×2
// 最后 osMessageQueueDelete(g_q)
```

## 5. 交通灯（红5s→绿5s→黄2s，蜂鸣间歇不同）

```c
// 初始化: GPIO10→PWM1, GPIO11→PWM2, GPIO12→PWM3, GPIO9→PWM0
// LedOn(pwm): 关所有 → Start(pwm,90,4000)

while (1) {
    LedOn(PWM1); BuzzerRun(500,500,5000);  // 红5s, 0.5s响停
    LedOn(PWM2); BuzzerRun(200,200,5000);  // 绿5s, 0.2s响停
    LedOn(PWM3); BuzzerRun(500,500,2000);  // 黄2s, 0.5s响停
}

// BuzzerRun(on,off,total): while(time<total){Start(0,20,4k);Delay(on);Stop(0);Delay(off);}
```

## 10. OLED滚动（学号+姓名+26字母，16字符窗口，500ms左移）

```c
ssd1306_Init();
char buf[256]; int len, off=0;
sprintf(buf, "%s%s%s", "学号", "姓名缩写", "abcdefghijklmnopqrstuvwxyz");
len = strlen(buf);

while (1) {
    char s[17]={0};
    for (int i=0;i<16;i++) s[i]=buf[(off+i)%len];
    ssd1306_Fill(Black); ssd1306_SetCursor(0,20);
    ssd1306_DrawString(s, Font_7x10, White); ssd1306_UpdateScreen();
    off++; osDelay(500);
}
```

## 11. 按键移字符（S1左移7px, S2右移7px, 边界停, 边沿检测）

```c
ssd1306_Init();
int x=(128-7)/2, y=(64-10)/2, last=KEY_NONE;  // 屏128×64, 字7×10

while (1) {
    int key = GetKey();  // 读按键
    if (key!=KEY_NONE && last==KEY_NONE) {     // 边沿：按下瞬间
        if (key==KEY_S1 && x>0)   x-=7;
        if (key==KEY_S2 && x<121) x+=7;        // 121=128-7
    }
    last = key;
    ssd1306_Fill(Black);
    ssd1306_DrawChar(x, y, 'x', Font_7x10, White);
    ssd1306_UpdateScreen();
    osDelay(50);
}
```

## 12. 交通灯倒计时OLED（绿20→黄5→红15，每秒刷新）

```c
ssd1306_Init();
int state=0;  // 0绿1黄2红

while (1) {
    int sec = (state==0)?20 : (state==1)?5 : 15;
    while (sec > 0) {
        char line[32];
        ssd1306_Fill(Black);
        ssd1306_SetCursor(20,20);
        ssd1306_DrawString((state==0)?"Green":(state==1)?"Yellow":"Red", Font_7x10, White);
        snprintf(line,32,"Time: %02d",sec);
        ssd1306_SetCursor(20,45); ssd1306_DrawString(line, Font_7x10, White);
        ssd1306_UpdateScreen();
        osDelay(1000); sec--;
    }
    state = (state+1)%3;
}
```
