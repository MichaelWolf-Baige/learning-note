## 第 1 题：互斥锁抢票

### 第一部分：死记硬背的骨架（每题照抄，只改红色字）

```c
#include <stdio.h>
#include <unistd.h>
#include "ohos_init.h"
#include "cmsis_os2.h"

static int g_ticket_count = 100;

// ---- newThread（固定，不用改）----
osThreadId_t newThread(char *name, osThreadFunc_t func, void *arg)
{
    osThreadAttr_t attr = { name, 0, NULL, 0, NULL, 1024*2, osPriorityNormal, 0, 0 };
    return osThreadNew(func, arg, &attr);
}
// ------------------------------------

// ③ 黄牛线程
void number_thread(void *arg)
{
    osMutexId_t *mid = (osMutexId_t *)arg;
    while (1) {
        osMutexAcquire(*mid, osWaitForever);
        if (g_ticket_count > 0) {
            g_ticket_count--;
            printf("抢到票，剩余%d\n", g_ticket_count);
            osMutexRelease(*mid);
            osDelay(200);          // ← 改这里：线程1:200  线程2:250  线程3:300
        } else {
            osMutexRelease(*mid);
            break;
        }
    }
}

// ② 主线程
void rtosv2_mutex_main_cjh_hys(void *arg)
{
    (void)arg;
    osMutexId_t mid = osMutexNew(&(osMutexAttr_t){0});
    newThread("cjh_hys1", number_thread, &mid);
    newThread("cjh_hys2", number_thread, &mid);
    newThread("cjh_hys3", number_thread, &mid);
    osDelay(30000);
}

// ① 入口
static void MutexTestTask(void)
{
    osThreadAttr_t attr = { .name="rtosv2_mutex_main_cjh_hys", .stack_size=1024, .priority=osPriorityNormal };
    osThreadNew((osThreadFunc_t)rtosv2_mutex_main_cjh_hys, NULL, &attr);
}
APP_FEATURE_INIT(MutexTestTask);
```

## 第 2 题：信号量工人

### 只改标红的部分，其余和第 1 题一样

```c
#include <stdio.h>
#include <unistd.h>
#include "ohos_init.h"
#include "cmsis_os2.h"

static osSemaphoreId_t semMachine;          // ← 全局信号量

// ---- newThread（和第1题完全一样）----
osThreadId_t newThread(char *name, osThreadFunc_t func, void *arg)
{
    osThreadAttr_t attr = { name, 0, NULL, 0, NULL, 1024*2, osPriorityNormal, 0, 0 };
    return osThreadNew(func, arg, &attr);
}

// ③ 工人线程
void worker_thread(void *arg)
{
    int *work_time = (int *)arg;
    while (1) {
        osSemaphoreAcquire(semMachine, osWaitForever);   // ← 获取
        printf("获得机器，工作%dms\n", *work_time);
        osDelay(*work_time);                             // ← 干活
        osSemaphoreRelease(semMachine);                  // ← 释放
        printf("归还机器\n");
        osDelay(100);   // 休息
    }
}

// ② 主线程
void rtosv2_semp_main_cjh_hys(void *arg)
{
    (void)arg;
    semMachine = osSemaphoreNew(3, 3, NULL);    // ← 3台机器
    int t1=200, t2=300, t3=400, t4=500;         // ← 四个时间
    newThread("cjh_hys1", worker_thread, &t1);
    newThread("cjh_hys2", worker_thread, &t2);
    newThread("cjh_hys3", worker_thread, &t3);
    newThread("cjh_hys4", worker_thread, &t4);
    osDelay(10000);
}

// ① 入口
static void SempTestTask(void)
{
    osThreadAttr_t attr = { .name="rtosv2_semp_main_cjh_hys", .stack_size=1024, .priority=osPriorityNormal };
    osThreadNew((osThreadFunc_t)rtosv2_semp_main_cjh_hys, NULL, &attr);
}
APP_FEATURE_INIT(SempTestTask);
```

## 第 3 题：消息队列

```c
#include <stdio.h>
#include <unistd.h>
#include "ohos_init.h"
#include "cmsis_os2.h"

typedef struct {
    char letter;
    int sender_id;
} Message_t;

static osMessageQueueId_t msgQueue;

// ---- newThread（完全相同）----
osThreadId_t newThread(char *name, osThreadFunc_t func, void *arg)
{
    osThreadAttr_t attr = { name, 0, NULL, 0, NULL, 1024*2, osPriorityNormal, 0, 0 };
    return osThreadNew(func, arg, &attr);
}

// ③ 发送者
void sender_thread(void *arg)
{
    int sender_id = *(int *)arg;
    Message_t msg;
    for (char c = 'a'; c <= 'z'; c++) {
        msg.letter = c;  msg.sender_id = sender_id;
        osMessageQueuePut(msgQueue, &msg, 0, osWaitForever);
        osDelay(100);
    }
}

// ③ 接收者
void receiver_thread(void *arg)
{
    int recv_id = *(int *)arg;
    Message_t msg;
    while (1) {
        osMessageQueueGet(msgQueue, &msg, NULL, osWaitForever);
        if (msg.letter == 'z') break;       // ← 收到 z 退出
    }
}

// ② 主线程
void rtosv2_msgq_main_cjh_hys(void *arg)
{
    (void)arg;
    msgQueue = osMessageQueueNew(10, sizeof(Message_t), NULL);  // ← 队列大小10
    int r1=1, r2=2, r3=3, s1=1, s2=2;
    newThread("Receiver1", receiver_thread, &r1);
    newThread("Receiver2", receiver_thread, &r2);
    newThread("Receiver3", receiver_thread, &r3);
    newThread("Sender1", sender_thread, &s1);
    newThread("Sender2", sender_thread, &s2);
    osDelay(5000);
    osMessageQueueDelete(msgQueue);
}

// ① 入口
static void MessageTestTask(void)
{
    osThreadAttr_t attr = { .name="rtosv2_msgq_main_cjh_hys", .stack_size=1024, .priority=osPriorityNormal };
    osThreadNew((osThreadFunc_t)rtosv2_msgq_main_cjh_hys, NULL, &attr);
}
APP_FEATURE_INIT(MessageTestTask);
```

## 硬件题 while(1) 里面写什么 —— 一张表

| 题号 | 题目 | while(1) 里写的核心逻辑 |
|------|------|------------------------|
| 4 | 按键蜂鸣器 | 读 GPIO5 → 按下(`==0`)则 `IoTPwmStart(0, 20, 4000)`，松开则 `IoTPwmStop(0)` + `osDelay(20)` |
| 5 | 交通灯 | `Red_On()` + `Buzzer_Run(500,500,5000)` → `Green_On()` + `Buzzer_Run(200,200,5000)` → `Yellow_On()` + `Buzzer_Run(500,500,2000)` |
| 6 | 炫彩灯 | 三重 for: `for(i=1;i<100;i++) for(j=1;j<100;j++) for(k=1;k<100;k++) { Start(1,i,4000); Start(2,j,4000); Start(3,k,4000); usleep(50000); Stop(1); Stop(2); Stop(3); }` |
| 7 | 小夜灯 | `hi_adc_read(CH3,&human,...); hi_adc_read(CH4,&light,...); if(light>1200 && human>1200) { PWM_Start RGB } else { PWM_Stop RGB } osDelay(100)` |
| 8 | 燃气报警 | `hi_adc_read(CH5,&data,...); if(data>600) IoTPwmStart(0,50,2700); else IoTPwmStop(0); osDelay(10)` |
