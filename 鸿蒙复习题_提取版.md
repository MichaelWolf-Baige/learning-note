### (4) 三线程抢票（互斥锁保护，票数 100，每次抢 1 张，耗时 200ms/250ms/300ms）

```c
#include <stdio.h>
#include <unistd.h>
#include "ohos_init.h"
#include "cmsis_os2.h"

static int g_ticket_count = 100;

void number_thread(void *arg)
{
    osMutexId_t *mid = (osMutexId_t *)arg;
    const char *name = osThreadGetName(osThreadGetId());
    while (1)
    {
        if (osMutexAcquire(*mid, osWaitForever) == osOK)
        {
            if (g_ticket_count > 0)
            {
                g_ticket_count--;
                printf("[Ticket] %s抢到一张票，剩余票数：%d\r\n",
                    osThreadGetName(osThreadGetId()), g_ticket_count);
                osMutexRelease(*mid);
                // 模拟抢票耗时
                if (name[3] == '1')      osDelay(200);
                else if (name[3] == '2') osDelay(250);
                else if (name[3] == '3') osDelay(300);
            }
            else
            {
                osMutexRelease(*mid);
                printf("[Ticket] %s票已售完，退出抢票\r\n",
                    osThreadGetName(osThreadGetId()));
                break;
            }
        }
    }
}
```

```c
#include <stdio.h>
#include <unistd.h>
#include "ohos_init.h"
#include "cmsis_os2.h"

void worker_thread(void *arg)
{
    int *work_time = (int *)arg;
    while (1)
    {
        if (osSemaphoreAcquire(semMachine, osWaitForever) == osOK)
        {
            printf("[Worker] %s获得一台机器,开始工作%dms\r\n",

                   osThreadGetName(osThreadGetId()));
```

```c
typedef struct {
    char letter;
    int sender_id;
} Message_t;

static osMessageQueueId_t msgQueue;

void rtosv2_msgq_main_cjh_hys(void *arg)
{

    {
        printf("[Queue] Create message queue failed.\r\n");
        return;
    }
    printf("[Queue] Create message queue success, id: %p, size: 10\r\n", msgQueue);
    int receiver_id1 = 1, receiver_id2 = 2, receiver_id3 = 3;
    int sender_id1 = 1, sender_id2 = 2;
    // 创建3个接收线程
    newThread("Receiver1", receiver_thread, &receiver_id1);
    newThread("Receiver2", receiver_thread, &receiver_id2);
    newThread("Receiver3", receiver_thread, &receiver_id3);
    // 创建2个发送线程
    newThread("Sender1", sender_thread, &sender_id1);
    newThread("Sender2", sender_thread, &sender_id2);
    osDelay(5000);
    osMessageQueueDelete(msgQueue);
    printf("[Queue] Message queue deleted.\r\n");
}
```

### (4) 发送者（a~z 循环发送）+ 接收者（收到 z 退出）

```c
// 发送者线程
void sender_thread(void *arg)
{
    int sender_id = *(int *)arg;
    Message_t msg;
    for (char letter = 'a'; letter <= 'z'; letter++)
    {
        msg.letter = letter;
        msg.sender_id = sender_id;
        if (osMessageQueuePut(msgQueue, &msg, 0, osWaitForever) == osOK)
        {
            printf("[Sender %d] 发送字母: %c\r\n", sender_id, letter);
        }
        osDelay(100);
    }
    printf("[Sender %d] 所有字母发送完毕，线程退出\r\n", sender_id);
}

// 接收者线程（收到 z 则退出）
void receiver_thread(void *arg)
{
    int receiver_id = *(int *)arg;
    Message_t msg;
    while (1)
    {
        if (osMessageQueueGet(msgQueue, &msg, NULL, osWaitForever) == osOK)
        {
            printf("[Receiver %d] 收到字母: %c (来自Sender %d)\r\n",
                receiver_id, msg.letter, msg.sender_id);
            if (msg.letter == 'z')
            {
                printf("[Receiver %d] 收到终止信号z，线程退出\r\n", receiver_id);
                break;
            }
        }
    }
}
```

### (2) 主线程：GPIO_5 按键按下 → GPIO_9 蜂鸣器响（占空比 20%，4000Hz）；松开 → 停止

> 辅助：按键消抖函数 `WaitKeyRelease`（检测按键松开）

```c
#define KEY_GPIO        5
#define BUZZER_GPIO     9
#define PWM_PORT        0
#define KEY_DEBOUNCE_MS 20

static void WaitKeyRelease(void)
{
    int last_state = IOT_GPIO_VALUE1;
    int stable_cnt = 0;
    while (1)
    {
        int value = 0;
        IoTGpioGetInputVal(KEY_GPIO, &value);
        if (value == last_state)
        {
            stable_cnt++;
            if (stable_cnt >= 5)
            {
                if (last_state == IOT_GPIO_VALUE1) break;
            }
        }
        else
        {
            last_state = value;
            stable_cnt = 0;
        }
        osDelay(KEY_DEBOUNCE_MS);
    }
}

static void PwmVolumeTask_cjh_hys(void *arg)
{
    (void)arg;
    int pwm_started = 0;
    // 初始化按键GPIO5
    IoTGpioInit(KEY_GPIO);
    hi_io_set_func(KEY_GPIO, HI_IO_FUNC_GPIO_5_GPIO);
    IoTGpioSetDir(KEY_GPIO, IOT_GPIO_DIR_IN);
    // 初始化蜂鸣器GPIO9 PWM输出
    IoTGpioInit(BUZZER_GPIO);
    hi_io_set_func(BUZZER_GPIO, HI_IO_FUNC_GPIO_9_PWM0_OUT);
    IoTGpioSetDir(BUZZER_GPIO, IOT_GPIO_DIR_OUT);
    IoTPwmInit(PWM_PORT);
    printf("[PwmVolumeTask] Started! Press GPIO_%d to buzz GPIO_%d (4kHz, duty=20%%)\n",
        KEY_GPIO, BUZZER_GPIO);

    while (1)
    {
        int current_state = IOT_GPIO_VALUE1;
        IoTGpioGetInputVal(KEY_GPIO, &current_state);
        int is_pressed = (current_state == IOT_GPIO_VALUE0);

        if (is_pressed && !pwm_started)
        {
            IoTPwmStart(PWM_PORT, 20, 4000);
            pwm_started = 1;
            printf("[PwmVolumeTask] Key PRESSED -> PWM started (duty=20%%, freq=4000Hz)\n");
        }
        else if (!is_pressed && pwm_started)
        {
            IoTPwmStop(PWM_PORT);
            pwm_started = 0;
            printf("[PwmVolumeTask] Key RELEASED -> PWM stopped\n");
        }
        osDelay(KEY_DEBOUNCE_MS);
    }
}
```

### (2) 交通灯循环：红灯(GPIO_10) 5s → 绿灯(GPIO_11) 5s → 黄灯(GPIO_12) 2s，蜂鸣器配合不同频率

- 红/黄灯亮 → 蜂鸣器 0.5s 响 0.5s 停
- 绿灯亮 → 蜂鸣器 0.2s 响 0.2s 停
- 每次变灯输出学号姓名

```c
// 宏定义
#define NAME_SHORT     "cjh_hys"
#define STUDENT_ID     "202310176058 202310176034"
#define NAME           "cjh hys"

// 引脚定义
#define LED_R    HI_IO_NAME_GPIO_10
#define LED_G    HI_IO_NAME_GPIO_11
#define LED_Y    HI_IO_NAME_GPIO_12
#define BUZ      HI_IO_NAME_GPIO_9
#define PWM_R    HI_PWM_PORT_PWM1
#define PWM_G    HI_PWM_PORT_PWM2
#define PWM_Y    HI_PWM_PORT_PWM3
#define PWM_BUZ  HI_PWM_PORT_PWM0

// LED 控制函数
void Buzzer_On(void)  { IoTPwmStart(PWM_BUZ, 20, 4000); }
void Buzzer_Off(void) { IoTPwmStop(PWM_BUZ); }

void All_Led_Off(void) { IoTPwmStop(PWM_R); IoTPwmStop(PWM_G); IoTPwmStop(PWM_Y); }
void Red_On(void)      { All_Led_Off(); IoTPwmStart(PWM_R, 90, 4000); }
void Green_On(void)    { All_Led_Off(); IoTPwmStart(PWM_G, 90, 4000); }
void Yellow_On(void)   { All_Led_Off(); IoTPwmStart(PWM_Y, 90, 4000); }

// 打印学号姓名
void Print_Info(void)
{
    printf("================================================\n");
    printf(" 学号: %s    姓名: %s\n", STUDENT_ID, NAME);
    printf("================================================\n");
}

// 蜂鸣器间歇运行
void Buzzer_Run(int on, int off, int total)
{
    int time = 0;
    while (time < total)
    {
        Buzzer_On();  osDelay(on);
        Buzzer_Off(); osDelay(off);
        time += on + off;
    }
}

// 主线程业务函数
static void TrafficLightTask_cjh_hys(void *arg)
{
    (void)arg;
    // 初始化 LED 硬件
    IoTGpioInit(LED_R); IoTGpioInit(LED_G); IoTGpioInit(LED_Y);

    IoTGpioSetDir(LED_Y, IOT_GPIO_DIR_OUT);
    IoTPwmInit(PWM_R); IoTPwmInit(PWM_G); IoTPwmInit(PWM_Y);
    // 初始化蜂鸣器
    IoTGpioInit(BUZ);
    hi_io_set_func(BUZ, HI_IO_FUNC_GPIO_9_PWM0_OUT);
    IoTPwmInit(PWM_BUZ);
    printf("交通灯系统启动成功！\n");

    while(1)
    {
        // 红灯5秒 + 蜂鸣器 0.5s响/0.5s停
        Print_Info();
        printf("状态: 红灯亮 -> 蜂鸣器 0.5s响 0.5s停\n");
        Red_On();
        Buzzer_Run(500, 500, 5000);
```

### 2. 时序循环：绿灯 20s → 黄灯 5s → 红灯 15s，无限循环

```c
#define GREEN_TIME   20
#define YELLOW_TIME  5
#define RED_TIME     15

while (1) {
    // 设置当前灯态和倒计时初始值
    switch (current_state) {
        case LIGHT_GREEN:  countdown = GREEN_TIME;  SetLight(LIGHT_GREEN);  break;
        case LIGHT_YELLOW: countdown = YELLOW_TIME; SetLight(LIGHT_YELLOW); break;
        case LIGHT_RED:    countdown = RED_TIME;    SetLight(LIGHT_RED);    break;
    }
    // 每秒刷新屏幕倒计时
    while (countdown > 0) {
        UpdateOLEDDisplay(current_state, countdown);
        osDelay(1000);
        countdown--;
    }
    // 状态切换
    switch (current_state) {
        case LIGHT_GREEN:  current_state = LIGHT_YELLOW; break;
        case LIGHT_YELLOW: current_state = LIGHT_RED;    break;
        case LIGHT_RED:    current_state = LIGHT_GREEN;  break;
    }
}
```
