```c
// A：全局变量
static int g_ticket_count = 100;

{
    osThreadAttr_t attr = { name,0,NULL,0,NULL,1024*2,osPriorityNormal,0,0 };
    return osThreadNew(func, arg, &attr);

void number_thread(void *arg)
{
    osMutexId_t *mid = (osMutexId_t *)arg;
    while (1) {
        osMutexAcquire(*mid, osWaitForever);
        if (g_ticket_count > 0) {
            g_ticket_count--;
            osMutexRelease(*mid);
```

## 第 2 题：信号量工人

```c
// A：全局变量
static osSemaphoreId_t semMachine;

// B：辅助函数（newThread 和上面一模一样）
osThreadId_t newThread(char *name, osThreadFunc_t func, void *arg)
{
    osThreadAttr_t attr = { name,0,NULL,0,NULL,1024*2,osPriorityNormal,0,0 };
    return osThreadNew(func, arg, &attr);
}

void worker_thread(void *arg)
{
    int *work_time = (int *)arg;
    while (1) {
        osSemaphoreAcquire(semMachine, osWaitForever);   // 占机器
        osDelay(*work_time);                              // 干活
        osSemaphoreRelease(semMachine);                   // 还机器
        osDelay(100);
    }
}

// C：初始化
semMachine = osSemaphoreNew(3, 3, NULL);    // 3台机器
int t1=200, t2=300, t3=400, t4=500;

// D：循环逻辑（写 while(1) 上面，然后删掉 while）
newThread("cjh_hys1", worker_thread, &t1);
newThread("cjh_hys2", worker_thread, &t2);
newThread("cjh_hys3", worker_thread, &t3);
newThread("cjh_hys4", worker_thread, &t4);
```

## 第 3 题：消息队列

```c
// A：全局变量
typedef struct { char letter; int sender_id; } Message_t;
static osMessageQueueId_t msgQueue;

// B：辅助函数（newThread 同上省略）

void sender_thread(void *arg)
{
    int id = *(int *)arg;
    Message_t msg;
    for (char c = 'a'; c <= 'z'; c++) {
        msg.letter = c;  msg.sender_id = id;
        osMessageQueuePut(msgQueue, &msg, 0, osWaitForever);
        osDelay(100);
    }
}

void receiver_thread(void *arg)
{
    int id = *(int *)arg;
    Message_t msg;
    while (1) {
        osMessageQueueGet(msgQueue, &msg, NULL, osWaitForever);
        if (msg.letter == 'z') break;
    }
}

// C：初始化
msgQueue = osMessageQueueNew(10, sizeof(Message_t), NULL);
int r1=1,r2=2,r3=3, s1=1,s2=2;

// D：循环逻辑（写 while(1) 上面，删 while）
newThread("Receiver1", receiver_thread, &r1);
newThread("Receiver2", receiver_thread, &r2);
newThread("Receiver3", receiver_thread, &r3);
newThread("Sender1", sender_thread, &s1);
newThread("Sender2", sender_thread, &s2);
osDelay(5000);
osMessageQueueDelete(msgQueue);

// 名字：
// 【入口名】= MessageTestTask
```

## 第 10 题：OLED 滚动

> **头文件**：基础 4 个 + `#include "ssd1306.h"` + `#include <string.h>`

```c
// A：全局
#define STU_ID   "你的学号"
#define NAME_PY  "liming"
char ad_buf[256];
int str_len;

// B：辅助函数
void CreateAdString(void)
{
    sprintf(ad_buf, "%s%Sabcdefghijklmnopqrstuvwxyz", STU_ID, NAME_PY);
    str_len = strlen(ad_buf);
}

void ShowScroll(int offset)
{
    char show_data[17] = {0};
    for (int i=0; i<16; i++)
        show_data[i] = ad_buf[(offset + i) % str_len];
    ssd1306_Fill(Black);
    ssd1306_SetCursor(0, 20);
    ssd1306_DrawString(show_data, Font_7x10, White);
    ssd1306_UpdateScreen();
}

// C：初始化
ssd1306_Init();
CreateAdString();

// D：循环逻辑
int offset = 0;
while (1) {
    ShowScroll(offset);
    offset++;
    osDelay(500);
}

// 名字：
// 【入口名】= Ssd1306ScrollDemo
// 【函数名】= ScrollDisplayTask
```

## 第 12 题：交通灯倒计时

> **头文件**：同第 10 题

```c
// A：宏定义
#define GREEN_TIME   20
#define YELLOW_TIME  5
#define RED_TIME     15

// B：辅助函数（OLED 两行显示）
void UpdateDisplay(char *state, int sec)
{
    ssd1306_Fill(Black);
    ssd1306_SetCursor(20, 20);
    ssd1306_DrawString(state, Font_7x10, White);
    char buf[32];
    snprintf(buf, sizeof(buf), "Time: %02d sec", sec);
    ssd1306_SetCursor(20, 45);
    ssd1306_DrawString(buf, Font_7x10, White);
    ssd1306_UpdateScreen();
}

// C：初始化
ssd1306_Init();

// D：循环逻辑
int state = 0;
while (1) {
    int countdown;
    if (state==0)      { countdown=20; }        // 绿灯
    else if (state==1) { countdown=5;  }        // 黄灯
    else               { countdown=15; }        // 红灯

    while (countdown > 0) {
        UpdateDisplay(state==0?"Green":state==1?"Yellow":"Red", countdown);
        osDelay(1000);
        countdown--;
    }
    state = (state + 1) % 3;    // 切下一个灯
}

// 名字：
// 【入口名】= 题目给的
// 【函数名】= 题目给的_cjh_hys
// 【线程名】= 题目给的_cjh_hys
```
