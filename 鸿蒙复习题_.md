```c
#include <stdio.h>
#include <unistd.h>
#include "ohos_init.h"
#include "cmsis_os2.h"

// ███ 第3题多写：全局 ███
typedef struct { char letter; int sender_id; } Message_t;
static osMessageQueueId_t g_queue;             // 消息队列句柄

// ███ 第3题多写：辅助函数 ███
osThreadId_t CreateThread(char *name, osThreadFunc_t func, void *arg)
{
    osThreadAttr_t attr = { name, 0, NULL, 0, NULL, 1024*2, osPriorityNormal, 0, 0 };
    return osThreadNew(func, arg, &attr);
}

void SenderTask(void *arg)
{
    int id = *(int *)arg;
    Message_t msg;
    for (char c='a'; c<='z'; c++) {
        msg.letter = c;
        msg.sender_id = id;
        osMessageQueuePut(g_queue, &msg, 0, osWaitForever);
        osDelay(100);
    }
}

void ReceiverTask(void *arg)
{
    int id = *(int *)arg;
    Message_t msg;
    while (1) {
        osMessageQueueGet(g_queue, &msg, NULL, osWaitForever);
        if (msg.letter == 'z') break;          // 终止条件
    }
}

// █████████ 骨架 █████████
static void QueueMain(void *arg)
{
    (void)arg;

    // ███ 第3题多写：初始化 ███
    g_queue = osMessageQueueNew(10, sizeof(Message_t), NULL);
    int r1=1, r2=2, r3=3, s1=1, s2=2;
    CreateThread("Receiver1", ReceiverTask, &r1);
    CreateThread("Receiver2", ReceiverTask, &r2);
    CreateThread("Receiver3", ReceiverTask, &r3);
    CreateThread("Sender1",   SenderTask,   &s1);
    CreateThread("Sender2",   SenderTask,   &s2);
    osDelay(5000);
    osMessageQueueDelete(g_queue);
}

// ▼▼▼ 骨架不变 ▼▼▼
static void QueueEntry(void)
{
    osThreadAttr_t attr;
    attr.name = "QueueMain";
```

```c
    // ███ 第12题多写：初始化 ███
    ssd1306_Init();

    // ███ 第12题多写：循环 ███
    int state = 0;   // 0=绿灯 1=黄灯 2=红灯
    while (1) {
        // 各状态持续时间
        int sec = (state == 0) ? 20 : (state == 1) ? 5 : 15;

        while (sec > 0) {
            char buf[32];
            ssd1306_Fill(Black);
            ssd1306_SetCursor(20, 20);
            ssd1306_DrawString(
                (state == 0) ? "Green" : (state == 1) ? "Yellow" : "Red",
                Font_7x10, White);
            snprintf(buf, sizeof(buf), "Time: %02d", sec);
            ssd1306_SetCursor(20, 45);
            ssd1306_DrawString(buf, Font_7x10, White);
            ssd1306_UpdateScreen();
            osDelay(1000);
            sec--;
        }
        state = (state + 1) % 3;
    }
}

static void TrafficCountdownEntry(void)
{
    osThreadAttr_t attr;
    attr.name = "TrafficCountdownMain";
    attr.stack_size = 4096;
    attr.priority = osPriorityNormal;
    osThreadNew(TrafficCountdownMain, NULL, &attr);
}
APP_FEATURE_INIT(TrafficCountdownEntry);
```
