## 1. 互斥锁 — 抢票

> 100张票，3线程抢，各线程速度不同。票为0退出。

```c
static int g_ticket = 100;

void number_thread(void *arg) {
    osMutexId_t *mid = (osMutexId_t *)arg;
    while (1) {
        osMutexAcquire(*mid, osWaitForever);   // 拿锁
        if (g_ticket > 0) {
            g_ticket--;                         // 抢一张
            osMutexRelease(*mid);               // 放锁
        } else {
            osMutexRelease(*mid);               // 没票了,放锁退出
            break;
        }
    }
}

// 主函数中:
osMutexId_t mid = osMutexNew(NULL);
newThread("xx1", number_thread, &mid);  // ×3
```

> **API链:** `osMutexNew` → `osMutexAcquire` → (临界区) → `osMutexRelease`

## 2. 信号量 — 工人用机器

> 3台机器(信号量初值3)，4个工人(线程)，工时200/300/400/500ms

```c
static osSemaphoreId_t g_sem;

void worker_thread(void *arg) {
    int *t = (int *)arg;
    while (1) {
        osSemaphoreAcquire(g_sem, osWaitForever);  // P: 占机器
        osDelay(*t);                                // 干活
        osSemaphoreRelease(g_sem);                  // V: 还机器
        osDelay(100);                               // 休息
    }
}

// 主函数中:
g_sem = osSemaphoreNew(3, 3, NULL);   // 最大3, 初始3
int t1=200, t2=300, t3=400, t4=500;
newThread("w1", worker_thread, &t1);  // ×4
```

> **API链:** `osSemaphoreNew(max,init)` → `osSemaphoreAcquire` → `osSemaphoreRelease`

## 3. 消息队列 — a~z 收发

> 队列长度10，2个发送者(a→z循环)，3个接收者(收到z退出)

```c
typedef struct { char letter; int sender_id; } Msg_t;
static osMessageQueueId_t g_q;

void sender_thread(void *arg) {
    int id = *(int*)arg;
    Msg_t m;
    for (char c = 'a'; c <= 'z'; c++) {
        m.letter = c; m.sender_id = id;
        osMessageQueuePut(g_q, &m, 0, osWaitForever);
        osDelay(100);
    }
}

void receiver_thread(void *arg) {
    Msg_t m;
    while (1) {
        osMessageQueueGet(g_q, &m, NULL, osWaitForever);
        if (m.letter == 'z') break;     // 收到z→退出
    }
}

// 主函数中:
g_q = osMessageQueueNew(10, sizeof(Msg_t), NULL);
newThread("R1", receiver_thread, &(int){1});  // 收×3
newThread("S1", sender_thread,   &(int){1});  // 发×2
osDelay(5000);
osMessageQueueDelete(g_q);
```

> **API链:** `osMessageQueueNew(len,size)` → `Put` / `Get`

## 4. 按键控制蜂鸣器

> GPIO5按键按下→GPIO9蜂鸣器响(PWM0, 20%占空比, 4000Hz)，松开→停

```c
// 初始化: GPIO5→IN, GPIO9→PWM0_OUT, IoTPwmInit(0)
while (1) {
    int v;
    IoTGpioGetInputVal(5, &v);
    if (v == IOT_GPIO_VALUE0)   // 按下=低电平
        IoTPwmStart(0, 20, 4000);
    else
        IoTPwmStop(0);
    osDelay(20);
}
```

## 5. 交通灯

> 红(GPIO10)5s → 绿(GPIO11)5s → 黄(GPIO12)2s 循环
> 红灯/黄灯时蜂鸣器0.5s响0.5s停，绿灯时0.2s响0.2s停

```c
// 初始化: 4个GPIO配PWM, IoTPwmInit(0~3)
// LedOn(pwm) = IoTPwmStart(pwm, 90, 4000)
// BuzzerRun(on, off, total): 循环(osDelay(on)响, osDelay(off)停)

while (1) {
    LedOn(PWM1); BuzzerRun(500, 500, 5000);   // 红5s, 0.5s间歇
    LedOn(PWM2); BuzzerRun(200, 200, 5000);   // 绿5s, 0.2s间歇
    LedOn(PWM3); BuzzerRun(500, 500, 2000);   // 黄2s, 0.5s间歇
}
```

## 6. RGB炫彩灯

> 三层嵌套循环 1~99，RGB三路PWM占空比全组合，50ms切换

```c
// 初始化: GPIO10/11/12→PWM1/2/3, IoTPwmInit(1~3)
while (1) {
    for (int i=1; i<100; i++)       // 红(R)
        for (int j=1; j<100; j++)   // 绿(G)
            for (int k=1; k<100; k++) { // 蓝(B, 黄灯)
                IoTPwmStart(1, i, 4000);
                IoTPwmStart(2, j, 4000);
                IoTPwmStart(3, k, 4000);
                usleep(50000);          // 50ms
                IoTPwmStop(1); IoTPwmStop(2); IoTPwmStop(3);
            }
}
```

## 7. 智能小夜灯

> ADC3(人体红外) + ADC4(光敏)，阈值均为1200
> 夜晚(光敏>1200) 且 有人(人体>1200) → RGB暖色灯亮，其余情况灭

```c
// 初始化: GPIO10/11/12→PWM LED (同第6题)
while (1) {
    unsigned short human = 0, light = 0;
    hi_adc_read(HI_ADC_CHANNEL_3, &human, HI_ADC_EQU_MODEL_4, HI_ADC_CUR_BAIS_DEFAULT, 0);
    hi_adc_read(HI_ADC_CHANNEL_4, &light, HI_ADC_EQU_MODEL_4, HI_ADC_CUR_BAIS_DEFAULT, 0);

    if (light > 1200 && human > 1200) {  // 暗 + 有人 → 亮
        IoTPwmStart(1, 65, 4000);   // 暖色RGB
        IoTPwmStart(2, 39, 4000);
        IoTPwmStart(3, 8, 4000);
    } else {                              // 灭灯
        IoTPwmStop(1); IoTPwmStop(2); IoTPwmStop(3);
    }
    osDelay(100);
}
```

## 9. 温湿度检测（AHT20）

> I2C0(GPIO13=SDA, GPIO14=SCL)接AHT20，湿度>90%蜂鸣器报警

```c
// I2C初始化: GPIO13→I2C0_SDA, GPIO14→I2C0_SCL
//             IoTI2cInit(0, 400000), AHT20_Calibrate()
// 蜂鸣器:     GPIO9→PWM0, IoTPwmInit(0)

float t, h;
while (1) {
    AHT20_StartMeasure();
    AHT20_GetMeasureResult(&t, &h);
    if (h > 90.0f) IoTPwmStart(0, 50, 2700);  // 湿度过高报警
    else           IoTPwmStop(0);
    osDelay(100);
}
```

## 10. OLED滚动显示

> 拼接"学号+姓名拼音+26字母a~z"，16字符窗口，500ms左移1格，环形滚动

```c
// 头文件加: ssd1306.h, string.h

char buf[256];
sprintf(buf, "%s%sabcdefghijklmnopqrstuvwxyz", "学号", "姓名拼音");
int len = strlen(buf), off = 0;

ssd1306_Init();
while (1) {
    char show[17] = {0};
    for (int i = 0; i < 16; i++)
        show[i] = buf[(off + i) % len];    // 环形取16字符
    ssd1306_Fill(Black);
    ssd1306_SetCursor(0, 20);
    ssd1306_DrawString(show, Font_7x10, White);
    ssd1306_UpdateScreen();
    off++;                                  // 左移一格
    osDelay(500);
}
```

## 11. 按键控制字符移动

> S1左移(步进7px)，S2右移(步进7px)，边界停止。Font_7x10字体，边沿检测防长按

```c
int x = (128 - 7) / 2;        // 屏幕居中 (128屏宽, 7字宽)
int y = (64 - 10) / 2;        // 垂直居中 (64屏高, 10字高)
int last_key = KEY_NONE;

ssd1306_Init();
while (1) {
    int key = GetKey();       // 读取按键状态
    if (key != KEY_NONE && last_key == KEY_NONE) {  // 边沿检测(按下瞬间)
        if (key == KEY_S1 && x > 0)    x -= 7;      // 左移
        if (key == KEY_S2 && x < 121)  x += 7;      // 右移(边界=128-7)
    }
    last_key = key;
    ssd1306_Fill(Black);
    ssd1306_DrawChar(x, y, 'x', Font_7x10, White);   // 绘制字符
    ssd1306_UpdateScreen();
    osDelay(50);
}
```

## 12. 交通灯倒计时（OLED）

> OLED显示两行：第一行状态(Green/Yellow/Red)，第二行倒计时数值
> 绿灯20s → 黄灯5s → 红灯15s 循环，每秒刷新

```c
int state = 0;  // 0=Green, 1=Yellow, 2=Red
ssd1306_Init();

while (1) {
    int sec = (state == 0) ? 20 : (state == 1) ? 5 : 15;
    while (sec > 0) {
        char line[32];
        ssd1306_Fill(Black);
        // 第一行: 状态
        ssd1306_SetCursor(20, 20);
        ssd1306_DrawString(
            (state == 0) ? "Green" : (state == 1) ? "Yellow" : "Red",
            Font_7x10, White);
        // 第二行: 倒计时
        snprintf(line, sizeof(line), "Time: %02d", sec);
        ssd1306_SetCursor(20, 45);
        ssd1306_DrawString(line, Font_7x10, White);
        ssd1306_UpdateScreen();
        osDelay(1000);
        sec--;
    }
    state = (state + 1) % 3;  // 切换: 0→1→2→0
}
```
